### LEPES 1: Migration skeleton

```
Cel:    Alembic 036 — chunked data-only backfill.
Fajl:   alembic/versions/036_association_mode_backfill.py (uj)
Forras: 01_PLAN/session_S73_v1_4_2_phase_1c_kickoff.md Day 4 (lines 126-148).

Skeleton:
  """Backfill intake_packages.association_mode for pre-Phase-1b rows.

  Revision ID: 036
  Revises: 035
  Create Date: 2026-05-15

  Phase 1c Day 4 — architect condition C5 part a.

  Strategy (per row where association_mode IS NULL):
      description_count = 0  → leave NULL        (valid: no descriptions)
      description_count = 1  → 'single_description'
      description_count = file_count > 0  → 'order'
      otherwise               → leave NULL + emit WARNING for ops review

  Safety:
      - pg_advisory_xact_lock() before the scan so concurrent writes don't race.
      - Chunked: 1000 package_ids per UPDATE — avoids long single transaction on
        large tables. Commits at the natural Alembic transaction boundary (the
        migration runs inside a transaction already).
      - description_count / file_count come from COUNT() subqueries on
        intake_descriptions / intake_files, keyed by package_id.
      - No PII emitted in the WARNING log (only package_id + counts + tenant_id).

  Downgrade:
      no-op (backfill is data; reversing would require 'UPDATE ... SET NULL'
      which is semantically destructive — documented here and in the PR body).
  """

  revision: str = "036"
  down_revision: str | None = "035"
  branch_labels = None
  depends_on = None

Implementation pattern (pseudocode):
  def upgrade() -> None:
      bind = op.get_bind()
      bind.execute(sa.text("SELECT pg_advisory_xact_lock(79_104_036)"))  # arbitrary lock id
      # Aggregate counts, chunk by package_id
      CHUNK = 1000
      last_id = None
      while True:
          query = sa.text("""
              WITH candidates AS (
                  SELECT p.package_id,
                         COALESCE(d.cnt, 0) AS description_count,
                         COALESCE(f.cnt, 0) AS file_count
                  FROM intake_packages p
                  LEFT JOIN (SELECT package_id, COUNT(*) AS cnt
                             FROM intake_descriptions GROUP BY package_id) d USING (package_id)
                  LEFT JOIN (SELECT package_id, COUNT(*) AS cnt
                             FROM intake_files GROUP BY package_id) f USING (package_id)
                  WHERE p.association_mode IS NULL
                    AND (:last IS NULL OR p.package_id > :last)
                  ORDER BY p.package_id
                  LIMIT :limit
              )
              SELECT package_id, description_count, file_count FROM candidates
          """)
          rows = bind.execute(query, {"last": last_id, "limit": CHUNK}).fetchall()
          if not rows:
              break
          for pkg_id, dc, fc in rows:
              mode = _decide_mode(dc, fc)
              if mode is not None:
                  bind.execute(
                      sa.text(
                          "UPDATE intake_packages SET association_mode = :m "
                          "WHERE package_id = :pid"
                      ),
                      {"m": mode, "pid": pkg_id},
                  )
              elif dc > 0 and fc != dc:
                  # ambiguous N/M: leave NULL, emit structured warning.
                  print(
                      f"WARNING association_mode_backfill_ambiguous "
                      f"package_id={pkg_id} description_count={dc} file_count={fc}"
                  )
          last_id = rows[-1][0]

  def _decide_mode(dc: int, fc: int) -> str | None:
      if dc == 0:
          return None
      if dc == 1:
          return "single_description"
      if dc == fc and fc > 0:
          return "order"
      return None

  def downgrade() -> None:
      # No-op: this is a data-only migration. See docstring.
      pass
```
